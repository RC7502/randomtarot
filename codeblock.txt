  function hashString(str) {
    // Deterministic 32-bit hash
    let h1 = 0xdeadbeef ^ str.length;
    let h2 = 0x41c6ce57 ^ str.length;

    for (let i = 0; i < str.length; i++) {
      const ch = str.charCodeAt(i);
      h1 = Math.imul(h1 ^ ch, 2654435761);
      h2 = Math.imul(h2 ^ ch, 1597334677);
    }

    h1 = Math.imul(h1 ^ (h1 >>> 16), 2246822507) ^ Math.imul(h2 ^ (h2 >>> 13), 3266489909);
    h2 = Math.imul(h2 ^ (h2 >>> 16), 2246822507) ^ Math.imul(h1 ^ (h1 >>> 13), 3266489909);

    return 4294967296 * (2097151 & h2) + (h1 >>> 0);
  }

  function getDateKey() {
    // Uses visitor's local date.
    // If you want a fixed timezone later, this is the part to change.
    const now = new Date();
    const year = now.getFullYear();
    const month = String(now.getMonth() + 1).padStart(2, "0");
    const day = String(now.getDate()).padStart(2, "0");
    return `${year}-${month}-${day}`;
  }

  function getCardForSignAndDate(sign, dateKey) {
    const seed = `${sign}|${dateKey}`;
    const hash = hashString(seed);

    const cardIndex = hash % CARDS.length;
    const reversed = ((Math.floor(hash / CARDS.length)) % 2) === 1;

    return {
      card: CARDS[cardIndex],
      reversed
    };
  }

  function buildImageUrl(card) {
    return BASE_IMAGE_URL + card.slug + IMAGE_EXTENSION;
  }

  function renderCard() {
    const sign = signSelect.value;
    const dateKey = getDateKey();
    const result = getCardForSignAndDate(sign, dateKey);

    const imageUrl = buildImageUrl(result.card);
    const labelText = result.reversed
      ? `(Reversed) ${result.card.name}`
      : result.card.name;

    cardImage.src = imageUrl;
    cardImage.alt = labelText;
    cardImage.style.transform = result.reversed ? "rotate(180deg)" : "rotate(0deg)";
    cardLabel.textContent = labelText;
  }

  signSelect.addEventListener("change", renderCard);

  renderCard();
